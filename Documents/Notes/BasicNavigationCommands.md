echo output-here = print what has been typed "output-here".
pwd = print the current working directory.
cat file-name.extension = Print file contents.
ls = list contents of the current directory.
cd directory-name = change into an existing directory.
cd .. = change backward; go back to the parent directory.
cd directory-name/sub-directory-name = change forward into multiple directories.
cd ../.. = Change backward on multiple directories.
clear = clear the command line.
mkdir: making new directories.
touch: create files.
echo "output-here" > file.txt = write output to file.
mv: moving files and directories.
rm and rmdir: removing files, removing directories.
cp: copying file/directory contents.
